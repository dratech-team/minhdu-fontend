interface BlockSalary {
  title: string,
  type: SalaryTypeEnum,
  rate?: {
    disabled?: boolean,
    show?: boolean
  }
  unit?: {
    units: {
      name: string,
      value: DatetimeUnitEnum,
    }[],
    disabled?: boolean
  },
  datetime?: {
    type: NzDateMode
    disabled?: boolean,
    show?: boolean
  },
  rangeDay?: {
    type: NzDateMode
    disabled?: boolean,
    show?: boolean
  },
  constraintHoliday?: {
    disabled?: boolean,
    show?: boolean
  },
  constraintOvertime?: {
    disabled?: boolean,
    show?: boolean
  }
  recipesConstant?: {
    name: string,
    value: PriceType
  }[],
  employeeConstant?: {
    name: string,
    value: EmployeeType
  }[],
  price?: {
    disabled?: boolean,
    show?: boolean
  }
  insurance?: {
    disabled?: boolean,
    show?: boolean
  }
  branch?: {
    disabled?: boolean,
    show: boolean
  },
  position?: {

    show: boolean
  },
  hasConstraints?: {
    disabled?: boolean,
    show: boolean
  },
  employeeType?: {
    disabled?: boolean,
    show: boolean
  }
}
